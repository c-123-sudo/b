#include <stdio.h> 
int max(int a, int b) { return a > b ? a : b; } 
int main() { 
  int w[] = {3, 5, 6, 2}, v[] = {10, 4, 9, 11}, n = 4, W = 7; 
  int dp[n+1][W+1]; 
  for (int i = 0; i <= n; i++) 
    for (int j = 0; j <= W; j++) 
      if (i == 0 || j == 0) dp[i][j] = 0; 
      else if (w[i-1] <= j) 
        dp[i][j] = max(v[i-1] + dp[i-1][j - w[i-1]], dp[i-1][j]); 
      else dp[i][j] = dp[i-1][j]; 
  printf("Max value: %d\n", dp[n][W]); 
}
